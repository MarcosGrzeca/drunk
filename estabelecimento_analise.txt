#Colégios

SELECT t.id, t.textoParserEmoticom, t.dataConvertida, tl.category, tl.name
FROM tweets t
JOIN tweet_localizacao tl ON tl.idTweetInterno = t.idInterno
WHERE q1 = 1
AND category IN ("College & University");


SELECT count(t.id), tl.name
FROM tweets t
JOIN tweet_localizacao tl ON tl.idTweetInterno = t.idInterno
WHERE q1 = 1
AND category IN ("College & University")
GROUP by tl.name
ORDER by 1 desc;

FROM tweets t
JOIN tweet_localizacao tl ON tl.idTweetInterno = t.idInterno
